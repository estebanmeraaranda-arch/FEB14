   Propiedades modificables:
   - width: 300px;           /* Ancho de la imagen */
   - height: 300px;          /* Alto de la imagen */

   - top: 20%;               /* Distancia desde la parte superior (o usa bottom:) */
   
   Ejemplos:
   - Para centro-derecha: left: 70%; top: 30%;
   - Para arriba-derecha: right: 5%; top: 5%;

*/
